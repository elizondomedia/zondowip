overflow: hidden; color: #fff; font-family: sans-serif; text-align: center; } /* Style for the start button and dropdown */ #micButton { margin: 20px; padding: 10px 20px; font-size: 1rem; } #gifSelector { margin: 20px; } /* Fade-in class (used after switching GIF) */ .fade-in { opacity: 1; } // List of all GIF filenames (ensure these files are in the same directory) const allGifs = [ "WHITE_000.GIF", "LIGHTBLUE_144.GIF", "LIGHTGREEN_244.GIF", "PINK_350.GIF", "TEAL_384.GIF", "BURGUNDY_396.GIF", "DARKGREEN_400.GIF", "DARKPURPLE_432.GIF", "DARKBLUE_480.GIF", "YELLOW_528.GIF", "TURQUOISE_540.GIF", "ORANGE_700.GIF", "LIGHTPURPLE_768.GIF", "SILVER_852.GIF", "GOLD_962.GIF", "gif 1 back.GIF", "gif 2 pulse.GIF", "gif 2 pulse2.GIF", "Werble-16F7860A03.GIF", "ZONDO GIF.GIF" ]; // Preload all GIF images to ensure smooth switching without delays const preloadedImages = {}; allGifs.forEach(src => { const img = new Image(); img.src = src; preloadedImages[src] = img; }); // Mapping from audio volume thresholds to corresponding GIFs (sorted ascending by threshold) const gifThresholds = [ { threshold: 0, file: "WHITE_000.GIF" }, { threshold: 144, file: "LIGHTBLUE_144.GIF" }, { threshold: 244, file: "LIGHTGREEN_244.GIF" }, { threshold: 350, file: "PINK_350.GIF" }, { threshold: 384, file: "TEAL_384.GIF" }, { threshold: 396, file: "BURGUNDY_396.GIF" }, { threshold: 400, file: "DARKGREEN_400.GIF" }, { threshold: 432, file: "DARKPURPLE_432.GIF" }, { threshold: 480, file: "DARKBLUE_480.GIF" }, { threshold: 528, file: "YELLOW_528.GIF" }, { threshold: 540, file: "TURQUOISE_540.GIF" }, { threshold: 700, file: "ORANGE_700.GIF" }, { threshold: 768, file: "LIGHTPURPLE_768.GIF" }, { threshold: 852, file: "SILVER_852.GIF" }, { threshold: 962, file: "GOLD_962.GIF" } ]; // Get references to DOM elements const gifImage = document.getElementById('gifDisplay'); const micButton = document.getElementById('micButton'); const gifSelector = document.getElementById('gifSelector'); // State variables let audioContext, analyser, microphoneStream; let autoMode = true; // whether GIF is controlled by mic (auto) or manually let currentGif = gifImage.src; // current displayed GIF filename let smoothedVolume = 0; // for smoothing the volume to avoid abrupt jumps // Function to smoothly transition to a new GIF image function showGif(newSrc) { if (newSrc === currentGif) return; // do nothing if it's already the current GIF // Fade out the current GIF, then change the source, then fade back in gifImage.style.opacity = 0; gifImage.onload = () => { // Once the new image is loaded, fade it in gifImage.style.opacity = 1; }; // Update the image source (will trigger onload when the image is ready) gifImage.src = newSrc; currentGif = newSrc; } // Function to analyze audio and update GIF in real time function analyzeAudio() { if (!autoMode || !analyser) { // If not in auto mode or audio not initialized, just loop again requestAnimationFrame(analyzeAudio); return; } // Get audio data (time domain waveform) const bufferLength = analyser.fftSize; const timeData = new Uint8Array(bufferLength); analyser.getByteTimeDomainData(timeData); // Compute volume as RMS (root mean square) of deviations from silence let sumSquares = 0; for (let i = 0; i < bufferLength; i++) { const deviation = timeData[i] - 128; sumSquares += deviation * deviation; } const rms = Math.sqrt(sumSquares / bufferLength); const volume = rms * 8; // scale up RMS to a 0-1024 range (approx) // Smooth the volume to prevent jittery rapid changes (low-pass filter) smoothedVolume = smoothedVolume * 0.8 + volume * 0.2; // Determine which GIF corresponds to the current smoothed volume let targetGif = gifThresholds[0].file; // default to the lowest (silence) for (let i = 0; i < gifThresholds.length; i++) { if (smoothedVolume >= gifThresholds[i].threshold) { targetGif = gifThresholds[i].file; } else { break; // since list is sorted, no need to check further once this fails } } // If the target GIF based on volume is different, switch to it with a transition if (targetGif !== currentGif && autoMode) { showGif(targetGif); } // Continue the animation loop requestAnimationFrame(analyzeAudio); } // Event handler: user selects a GIF from the dropdown gifSelector.addEventListener('change', function() { const selectedValue = this.value; if (selectedValue === "AUTO") { // Switch back to microphone-controlled mode autoMode = true; } else { // Switch to manual mode and display the selected GIF autoMode = false; showGif(selectedValue); } }); // Event handler: user clicks the Start Microphone button micButton.addEventListener('click', function() { if (audioContext) { return; // Already running } // Ask for microphone access navigator.mediaDevices.getUserMedia({ audio: true, video: false }) .then(stream => { // Create AudioContext and analyser audioContext = new (window.AudioContext || window.webkitAudioContext)(); analyser = audioContext.createAnalyser(); analyser.fftSize = 1024; // FFT size for audio analysis (controls resolution) const source = audioContext.createMediaStreamSource(stream); source.connect(analyser); // Start the analysis loop analyzeAudio(); }) .catch(err => { console.error('Microphone access denied or error:', err); }); });
